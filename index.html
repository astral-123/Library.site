Loading the Eclipse Library

local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/astral-123/Library/refs/heads/main/Library.lua"))()

Creating a Window

local Window = Library:CreateWindow({
    Name = "Eclipse Example Window",
    Icon = 10723407389, -- Lucide icon ID or 0 for no icon
    LoadingTitle = "Eclipse Interface Suite",
    LoadingSubtitle = "by Eclipse",
    Theme = "Default", -- Options: Default, Amethyst, Ocean, Rose, Green
    ToggleUIKeybind = "V", -- Key to toggle UI
    ConfigurationSaving = {
        Enabled = false,
        FolderName = nil,
        FileName = "EclipseHub"
    }
})
