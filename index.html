Loading the Nebula Library

local NebulaUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/astral-123/Library/refs/heads/main/Library.lua"))()

Creating a Window

local Window = NebulaUI:CreateWindow({
    Name = "Nebula Hub",
    ToggleKey = Enum.KeyCode.V,
    KeySystem = false,
    Key = "NebulaHub2024",
    Resizable = true
})

Notifying the user

Window:Notification({
    Title = "Welcome!",
    Text = "Nebula Hub loaded!",
    Duration = 3
})

Creating a Tab

local Main = Window:CreateTab("Main")

Creating a Button

local LeftSection = Main:AddSection("Main Functions", "left")

LeftSection:AddButton({
Name = "Start Farming",
Callback = function()
print("Farming started!")
end
})

Creating a Toggle

local LeftSection = Main:AddSection("Main Functions", "left")

LeftSection:AddToggle({
Name = "Auto Farm",
Default = false,
Callback = function(value)
print("Auto Farm:", value)
end
})

Creating a Slider

local RightSection = Main:AddSection("Settings", "right")

RightSection:AddSlider({
Name = "Speed",
Min = 100,
Max = 500,
Default = 300,
Callback = function(value)
print("Speed:", value)
end
})

Creating a Input

local Mainleft = Main:AddSection("Input", "left")

Mainleft:AddInput({
    Name = "Input",
    PlaceholderText = "Type here...",
    Flag = "Input1",
    Callback = function(text)
        print("Input:", text)
    end
})

Creating a Dropdown

local MainRight = Main:AddSection("Dropdown", "right")

MainRight:AddDropdown({
    Name = "Target Part",
    Options = {"Head", "UpperTorso", "LowerTorso", "HumanoidRootPart", "Random"},
    Default = "Head",
    Callback = function(Option)
        settings.aimPart = Option
        Window:Notification({
            Title = "Target Part",
            Text = "Target: " .. Option,
            Duration = 2
        })
    end
})
