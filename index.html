Library

Loading the Nebula Library

local NebulaUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/astral-123/Library/refs/heads/main/Library.lua"))()

Creating a Window

local Window = NebulaUI:CreateWindow({
    Name = "Nebula Hub",
    ToggleKey = Enum.KeyCode.V,
    KeySystem = false,      -- Active le système de clé
    Key = "NebulaHub2024",  -- La clé correcte
    Resizable = true       -- Active le redimensionnement
})

Creating a Tab

local Main = Window:CreateTab("Main")

Creating a Button

local LeftSection = Main:AddSection("Main Functions", "left")

LeftSection:AddButton({
    Name = "Start Farming",
    Callback = function()
        print("Farming started!")
    end
})

Creating a Toggle

local LeftSection = Main:AddSection("Main Functions", "left")

LeftSection:AddToggle({
    Name = "Auto Farm",
    Default = false,
    Callback = function(value)
        print("Auto Farm:", value)
    end
})

Creating a Slider

local RightSection = Main:AddSection("Settings", "right")

RightSection:AddSlider({
    Name = "Speed",
    Min = 100,
    Max = 500,
    Default = 300,
    Callback = function(value)
        print("Speed:", value)
    end
})

Creating a Dropdown

local RightSection = Main:AddSection("Settings", "right")

RightSection:AddDropdown({
    Name = "Weapon",
    Options = {"Melee", "Sword", "Gun"},
    Default = "Melee",
    Callback = function(value)
        print("Weapon:", value)
    end
})

Creating a TextBox

local RightSection = Main:AddSection("Settings", "right")

RightSection:AddTextBox({
    Name = "Player Name",
    Placeholder = "Enter player name...",
    Default = "",
    Callback = function(text, enterPressed)
        print("Player Name:", text)
        if enterPressed then
            print("Enter was pressed!")
        end
    end
})
